470 bool cyclicManagedMemory::checkCycle() const
471 {
472  rambrain_pthread_mutex_lock ( &stateChangeMutex );
473  rambrain_pthread_mutex_lock ( &cyclicTopoLock );
474 #ifdef PARENTAL_CONTROL
475  unsigned int no_reg = memChunks.size() - 1;
476 #else
477  unsigned int no_reg = memChunks.size();
478 #endif
479  unsigned int encountered = 0;
480  cyclicAtime *cur = active;
481  cyclicAtime *oldcur;
482 
483  if ( !cur ) {
484  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
485  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
486  if ( no_reg == 0 && counterActive == NULL ) {
487  return true;
488  } else {
489  return false;
490  }
491  }
492 
493  bool inActiveOnlySection = ( active->chunk->status == MEM_SWAPPED || active->chunk->status == MEM_SWAPOUT ? false : true );
494  bool inSwapsection = true;
495  do {
496  ++encountered;
497  oldcur = cur;
498  cur = cur->next;
499  if ( oldcur != cur->prev ) {
500  errmsgf ( "Mutual connecion failure at chunks %lu and %lu", oldcur->chunk->id, cur->chunk->id );
501  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
502  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
503 
504  return false;
505  }
506 
507  if ( inActiveOnlySection ) {
508  if ( oldcur->chunk->status == MEM_SWAPPED || oldcur->chunk->status == MEM_SWAPOUT ) {
509  errmsg ( "Swapped elements in active section!" );
510 
511  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
512  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
513  return false;
514  }
515  } else {
516  if ( oldcur->chunk->status == MEM_SWAPPED && !inSwapsection ) {
517  errmsg ( "Isolated swapped element block not tracked by counterActive found!" );
518  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
519  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
520  return false;
521  }
522  if ( oldcur->chunk->status != MEM_SWAPPED && oldcur->chunk->status != MEM_SWAPOUT ) {
523  inSwapsection = false;
524  }
525  }
526 
527  if ( oldcur == counterActive ) {
528  inActiveOnlySection = false;
529  }
530 
531  } while ( cur != active );
532  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
533  rambrain_pthread_mutex_unlock ( &stateChangeMutex );
534  if ( encountered != no_reg ) {
535  errmsgf ( "Not all elements accessed. %d expected, %d encountered", no_reg, encountered );
536  return false;
537  } else {
538  return true;
539  }

605 {
606 
607  rambrain_pthread_mutex_lock ( &cyclicTopoLock );
608  if ( counterActive == 0 ) {
609  rambrain_pthread_mutex_unlock ( &stateChangeMutex );

612  }
613  VERBOSEPRINT ( "swapOutEntry" );
614  if ( min_size > memory_max ) {
615  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
616  return ERR_MORETHANTOTALRAM;
617  }
618  global_bytesize swap_free = swap->getFreeSwap();
619  if ( swap_free < min_size ) {
620  if ( !swap->extendSwapByPolicy ( min_size ) ) {

622  return ERR_SWAPFULL;

634  cyclicAtime *countPos = counterActive;

636  global_bytesize unload_size2 = 0;

638  //In case of preemptives, we cannot break at active, but will have to go through preemptive elements to find swap candidates.
639  global_bytesize preemptiveBytesStill = preemptiveBytes;

652 #endif
653  }
654  }

660  }

662 #ifdef CYCLIC_VERBOSE_DBG
663  printf ( "emergency, once round!\n" );
664 #endif
665  break;
666  }
667 
668  countPos = countPos->prev;
669  }
670  if ( unload_size == 0 ) {
671  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
672  return ERR_NOTENOUGHCANDIDATES;
673  }
674 
675  managedMemoryChunk *unloadlist[unload];
676  managedMemoryChunk **unloadElem = unloadlist;
677 #ifdef CYCLIC_VERBOSE_DBG
678  printf ( "active = %d\n", active->chunk->id );
679 #endif
680  preemptiveBytesStill = preemptiveBytes;
681  activeReached = false;
682  while ( unload_size2 < unload_size ) {
683  ++passed;
684  if ( fromPos->chunk->status == MEM_ALLOCATED && ( unload_size2 + fromPos->chunk->size <= swap_free ) && ( fromPos->chunk->useCnt == 0 ) ) {
685  if ( fromPos->chunk->size + unload_size2 <= swap_free ) {
686  unload_size2 += fromPos->chunk->size;
687  *unloadElem = fromPos->chunk;
688  ++unloadElem;
689 #ifdef VERYVERBOSE
690  printf ( "swapout %d\n", fromPos->chunk->id );
691 #endif
692  if ( fromPos->chunk->preemptiveLoaded ) { //We had this chunk preemptive, but now have to swap out.
693  //This is a bit evil, as we will reload preemptive bytes when we've swapped them out.
695  performance ( too much preemptive action possible ). Naively testing, this is not the case.
696  fromPos->chunk->preemptiveLoaded = false;
697  preemptiveBytes -= fromPos->chunk->size;
698  }
699  }
700  }
701  preemptiveBytesStill -= fromPos->chunk->preemptiveLoaded ? fromPos->chunk->size : 0;
702  if ( fromPos == active ) {
703  activeReached = true;
704  }
705  if ( preemptiveBytesStill == 0 && activeReached ) {
706 #ifdef CYCLIC_VERBOSE_DBG
707  printf ( "emergency, once round!\n" );
708 #endif
709  break;
710  }
711 
712  fromPos = fromPos->prev;
713  }
714  global_bytesize real_unloaded = swap->swapOut ( unloadlist, unload );
715  bool swapSuccess = ( real_unloaded == unload_size2 ) ;
716  if ( !swapSuccess ) {
717  if ( real_unloaded == 0 ) {
718  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
719  return ERR_NOTENOUGHCANDIDATES;
720  }
721  }
722 #ifdef CYCLIC_VERBOSE_DBG
723  printf ( "active = %d\n", active->chunk->id );
724 #endif
725  fromPos = counterActive;
726  cyclicAtime *moveEnd, *cleanFrom;
727  moveEnd = NULL;
728 
729  bool inSwappedSection = ( fromPos->chunk->status == MEM_SWAPPED || fromPos->chunk->status == MEM_SWAPOUT );
730  bool doRoundtrip = fromPos == countPos;
731 
732 
733  cleanFrom = counterActive->next;
734 
735  while ( fromPos != countPos || doRoundtrip ) {
736 #ifdef CYCLIC_VERBOSE_DBG
737  printf ( "at %d\t", fromPos->chunk->id );
738 #endif
739  doRoundtrip = false;
740  if ( inSwappedSection ) {
741  if ( fromPos->chunk->status != MEM_SWAPPED && fromPos->chunk->status != MEM_SWAPOUT ) {
742  inSwappedSection = false;
743  if ( moveEnd ) {
744  // xxxxxxxxxxoooooooxxxxxxooooooo
745  // ------A--><--B--><-C--><--D
746  // Change order from A-B-C-D to A-C-B-D

754 #ifdef CYCLIC_VERBOSE_DBG

769  }
770  }

773  inSwappedSection = true;

775  moveEnd = fromPos;
776  }
777  }
778  }

780  }

783 #endif
784  if ( moveEnd ) {
785 
786  // xxxxxxxxxxoooooooxxxxxxooooooo
787  // ------A--><--B--><-C--><--D
788  // Change order from A-B-C-D to A-C-B-D
789  cyclicAtime *endNonswap = fromPos; //A
790  cyclicAtime *startIsoswap = fromPos->next; //B
791  cyclicAtime *endIsoswap = moveEnd; //B
792  cyclicAtime *startIsoNonswap = moveEnd->next;//C
793  cyclicAtime *endIsoNonswap = cleanFrom->prev;//C
794  cyclicAtime *startClean = cleanFrom;//D
795 #ifdef CYCLIC_VERBOSE_DBG
796  printf ( "LAST----%d><%d-%d><%d-%d><%d---\n", endNonswap->chunk->id, startIsoswap->chunk->id, endIsoswap->chunk->id,
797  startIsoNonswap->chunk->id, endIsoNonswap->chunk->id, startClean->chunk->id );
798 #endif

801  MUTUAL_CONNECT ( endIsoNonswap, startIsoswap );
802  MUTUAL_CONNECT ( endIsoswap, startClean );
803  cleanFrom = startIsoswap;
804  moveEnd = NULL;
805  }
806  counterActive = cleanFrom->prev;
807  while ( ! ( counterActive->chunk->status & MEM_ALLOCATED || counterActive->chunk->status == MEM_SWAPIN || counterActive == active ) ) {
808  counterActive = counterActive->prev;
809  }
810  if ( ! ( active->chunk->status & MEM_ALLOCATED || active->chunk->status == MEM_SWAPIN ) ) { // We may have swapped out the first allocated element
811  active = counterActive;
812 #ifdef CYCLIC_VERBOSE_DBG
813  printf ( "had to move active", fromPos->chunk->id );
814 #endif
815  }
816  rambrain_pthread_mutex_unlock ( &cyclicTopoLock );
817  VERBOSEPRINT ( "swapOutReturn" );
818  if ( swapSuccess ) {
819 #ifdef SWAPSTATS
820  swap_out_scheduled_bytes += unload_size;
821  n_swap_out += 1;
822 #endif
823  return ERR_SUCCESS;
824  } else {
825  if ( real_unloaded > min_size ) {
826  return ERR_SUCCESS;
827  } else {
828  return ERR_NOTENOUGHCANDIDATES;
829  }
830  }
